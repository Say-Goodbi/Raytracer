30 Color lerpColor(const Color &a, const Color &b, double t)
31 {
32 return Color(
33 a.r + (b.r - a.r) * t,
34 a.g + (b.g - a.g) * t,
35 a.b + (b.b - a.b) * t);
36 }

38 Color checksToHeatColor(double normalized)
39 {
40 const double t = std::clamp(normalized, 0.0, 1.0);
41
42 // Use a longer gray ramp before transitioning into colors.
43 // First part: black -> dark gray -> light gray (grayThreshold portion)
44 // Second part: color ramp (remaining portion)
45 const double grayThreshold = 0.6; // 60% of the ramp reserved for grays
46
47 if (t <= 0.0)
48 return Color(0.0, 0.0, 0.0);
49
50 if (t < grayThreshold) {
51 // interpolate from black -> dark gray -> light gray across [0, grayThreshold]
52 double gt = t / grayThreshold; // normalized within gray segment [0,1]
53 if (gt < 0.5) {
54 // black -> dark gray (first half)
55 double localT = gt / 0.5;
56 return lerpColor(Color(0.0,0.0,0.0), Color(0.2,0.2,0.2), localT);
57 } else {
58 // dark gray -> light gray (second half)
59 double localT = (gt - 0.5) / 0.5;
60 return lerpColor(Color(0.2,0.2,0.2), Color(0.85,0.85,0.85), localT);
61 }
62 }
63
64 // Color ramp portion mapped from [grayThreshold, 1.0] -> [0,1]
65 double ct = (t - grayThreshold) / (1.0 - grayThreshold);
66
67 const Color colorStops[] = {
68 Color(1.0, 1.0, 0.0), // yellow
69 Color(1.0, 0.5, 0.0), // orange
70 Color(1.0, 0.0, 0.0), // red
71 Color(0.6, 0.0, 1.0), // violet
72 Color(0.0, 0.0, 1.0) // blue
73 };
74 const int colorSegments = 4; // stops-1
75 double scaled = ct * colorSegments;
76 int idx = static_cast<int>(std::floor(scaled));
77 if (idx < 0) idx = 0;
78 if (idx >= colorSegments) idx = colorSegments - 1;
79 double localT = scaled - idx;
80 return lerpColor(colorStops[idx], colorStops[idx + 1], localT);
81 }

83 std::optional<Geometry::HitRecord> traverseNodeWithStats(const BVH::Node *node, const Geometry::Ray &ray, TraversalStats &stats)
84 {
85 if (!node)
86 return std::nullopt;
87
88 ++stats.nodeChecks;
89 if (!node->bounds.hit(ray))
90 return std::nullopt;
91
92 std::optional<Geometry::HitRecord> closest;
93
94 if (node->isLeaf())
95 {
96 for (const auto &primitive : node->primitives)
97 {
98 ++stats.primitiveChecks;
99 std::optional<Geometry::HitRecord> hit = primitive->hit(ray);
100 if (hit && (!closest || hit->rayDistance < closest->rayDistance))
101 closest = hit;
102 }
103 return closest;
104 }
105
106 std::optional<Geometry::HitRecord> leftHit = traverseNodeWithStats(node->left.get(), ray, stats);
107 std::optional<Geometry::HitRecord> rightHit = traverseNodeWithStats(node->right.get(), ray, stats);
108
109 if (leftHit && rightHit)
110 return leftHit->rayDistance < rightHit->rayDistance ? leftHit : rightHit;
111 return leftHit ? leftHit : rightHit;
112 }

114 std::optional<Geometry::HitRecord> traverseSceneWithStats(Scene &scene, const Geometry::Ray &ray, TraversalStats &stats)
115 {
116 std::optional<Geometry::HitRecord> closest;
117
118 std::optional<std::reference_wrapper<const BVH>> bvhRef = scene.getBVH();
119 if (bvhRef)
120 closest = traverseNodeWithStats(bvhRef->get().root(), ray, stats);
121
122 for (const auto &primitive : scene.getPrimitives())
123 {
124 if (primitive->getBounds())
125 continue;
126
127 ++stats.primitiveChecks;
128 std::optional<Geometry::HitRecord> hit = primitive->hit(ray);
129 if (hit && (!closest || hit->rayDistance < closest->rayDistance))
130 closest = hit;
131 }
132
133 return closest;
134 }

137 BVHCam::BVHCam(Geometry::Point3D pos, Geometry::Vector3D direction, float fov, int width, int height)
138 : ARenderer(width, height), _fov(fov)
139 {
140 // Build a transform matrix from position + forward direction.
141 Geometry::Vector3D forward = (direction.length() > 1e-6)
142 ? direction.normalize()
143 : Geometry::Vector3D(0, 1, 0); // default: look toward +y
144
145 Geometry::Vector3D worldUp = (std::abs(forward.z) > 0.9)
146 ? Geometry::Vector3D(0, 1, 0)
147 : Geometry::Vector3D(0, 0, 1); // z-up unless looking straight up/down
148
149 Geometry::Vector3D right = Geometry::Vector3D::cross(forward, worldUp).normalize();
150 Geometry::Vector3D screenUp = Geometry::Vector3D::cross(right, forward).normalize();
151
152 Geometry::TransformMatrix t = Geometry::TransformMatrix::identity();
153 // columns: right, up, forward
154 t(0,0) = right.x; t(1,0) = right.y; t(2,0) = right.z;
155 t(0,1) = screenUp.x; t(1,1) = screenUp.y; t(2,1) = screenUp.z;
156 t(0,2) = forward.x; t(1,2) = forward.y; t(2,2) = forward.z;
157 // translation
158 t(0,3) = pos.x; t(1,3) = pos.y; t(2,3) = pos.z;
159
160 _transform = t;
161 buildScreenFromTransform();
162 }

170 /**
171 * @brief Build a world-space ray from normalized screen coordinates.
172 *
173 * Maps (u, v) ∈ [0, 1]² onto the screen rectangle in world space, then
174 * returns a ray from the camera position toward that point.
175 *
176 * @param u Horizontal position (0 = left, 1 = right).
177 * @param v Vertical position (0 = top, 1 = bottom).
178 * @return Unit-direction ray from the camera through the screen point.
179 */
180 Geometry::Ray BVHCam::generateRay(double u, double v)
181 {
182 Geometry::Point3D screenPoint = this->_screen.pointAt(u, v);
183 Geometry::Point3D origin = this->_transform * Geometry::Point3D(0, 0, 0);
184 Geometry::Vector3D rayDirection = Geometry::Vector3D(
185 screenPoint.x - origin.x,
186 screenPoint.y - origin.y,
187 screenPoint.z - origin.z)
188 .normalize();
189 return Geometry::Ray(origin, rayDirection);
190 }

192 void BVHCam::buildScreenFromTransform()
193 {
194 // Extract basis columns and translation from _transform
195 Geometry::Point3D pos(
196 _transform(0,3),
197 _transform(1,3),
198 _transform(2,3));
199
200 Geometry::Vector3D right(
201 _transform(0,0),
202 _transform(1,0),
203 _transform(2,0));
204
205 Geometry::Vector3D up(
206 _transform(0,1),
207 _transform(1,1),
208 _transform(2,1));
209
210 Geometry::Vector3D forward(
211 _transform(0,2),
212 _transform(1,2),
213 _transform(2,2));
214
215 const double rightScale = std::max(right.length(), 1e-6);
216 const double upScale = std::max(up.length(), 1e-6);
217
218 // Normalize basis vectors (in case transform wasn't orthonormal)
219 forward = (forward.length() > 1e-6) ? forward.normalize() : Geometry::Vector3D(0, 1, 0);
220 right = (right.length() > 1e-6) ? right.normalize() : Geometry::Vector3D::cross(forward, Geometry::Vector3D(0,0,1)).normalize();
221 up = Geometry::Vector3D::cross(right, forward).normalize();
222
223 double fovRad = _fov * M_PI / 180.0;
224 double halfW = std::tan(fovRad / 2.0) * rightScale;
225 double halfH = std::tan(fovRad / 2.0) * ((double)_height / _width) * upScale;
226
227 // Screen center is 1 unit in front of the camera
228 Geometry::Point3D center(pos.x + forward.x, pos.y + forward.y, pos.z + forward.z);
229
230 // Top-left corner: move left by halfW and up by halfH
231 _screen.origin = Geometry::Point3D(
232 center.x - right.x * halfW + up.x * halfH,
233 center.y - right.y * halfW + up.y * halfH,
234 center.z - right.z * halfW + up.z * halfH);
235
236 _screen.width = Geometry::Vector3D(right.x * 2*halfW, right.y * 2*halfW, right.z * 2*halfW);
237
238 _screen.height = Geometry::Vector3D(-up.x * 2*halfH, -up.y * 2*halfH, -up.z * 2*halfH);
239 }

241 /**
242 * @brief Render the scene into a 2D framebuffer.
243 *
244 * Each pixel is super-sampled with a deterministic stratified grid of
245 * `_sqrtSamples × _sqrtSamples` sub-pixel rays. Sample positions are
246 * fixed at the **center of each sub-cell** — no jitter, no random offset —
247 * keeping the overall render fully deterministic.
248 *
249 * Per-pixel steps:
250 * 1. Iterate over the `_sqrtSamples × _sqrtSamples` sub-cell grid.
251 * 2. Compute the sub-cell center in normalized screen space (u, v).
252 * 3. Generate and path-trace a ray via `generateRay()` + `castRay()`.
253 * 4. Average all sub-pixel radiance values and clamp to [0, 1].
254 *
255 * Output layout: `frame[y][x]`, y = 0 at the top, x = 0 at the left.
256 *
257 * @param scene Scene to render.
258 * @return 2D grid of clamped colors: `frame[height][width]`.
259 */
260 std::vector<std::vector<Color>> BVHCam::render(Scene& scene, std::vector<std::vector<Color>>& framebuffer) {
261 scene.prepareAccelerationStructure();
262
263 {
264 std::unique_lock<std::shared_mutex> lock(FrameBufferMutex());
265 // Ensure framebuffer is the correct size before incremental updates start.
266 if ((int)framebuffer.size() != _height) framebuffer.assign(_height, std::vector<Color>(_width));
267 else {
268 for (auto &row : framebuffer) {
269 if ((int)row.size() != _width) row.assign(_width, Color(0,0,0));
270 }
271 }
272 }
273
274 std::vector<double> checksPerPixel(static_cast<std::size_t>(_width) * static_cast<std::size_t>(_height), 0.0);
275 double maxChecks = 0.0;
276 double minChecks = std::numeric_limits<double>::infinity();
277
278 for (int y = 0; y < _height; y++) {
279 std::vector<Color> row(_width, Color(0, 0, 0));
280 for (int x = 0; x < _width; x++) {
281 double accumulatedChecks = 0.0;
282
283 for (int sy = 0; sy < _sqrtSamples; sy++) {
284 for (int sx = 0; sx < _sqrtSamples; sx++) {
285 double u = (x + (sx + 0.5) / _sqrtSamples) / (_width - 1);
286 double v = (y + (sy + 0.5) / _sqrtSamples) / (_height - 1);
287 Geometry::Ray ray = generateRay(u, v);
288
289 TraversalStats stats;
290 (void)traverseSceneWithStats(scene, ray, stats);
291 accumulatedChecks += static_cast<double>(stats.nodeChecks + stats.primitiveChecks);
292 }
293 }
294
295 int totalSamples = _sqrtSamples * _sqrtSamples;
296 const double averageChecks = accumulatedChecks / static_cast<double>(totalSamples);
297 checksPerPixel[static_cast<std::size_t>(y) * static_cast<std::size_t>(_width) + static_cast<std::size_t>(x)] = averageChecks;
298 maxChecks = std::max(maxChecks, averageChecks);
299 minChecks = std::min(minChecks, averageChecks);
300 row[x] = Color(0, 0, 0);
301 }
302
303 {
304 std::unique_lock<std::shared_mutex> lock(FrameBufferMutex());
305 framebuffer[y] = std::move(row);
306 }
307 }
308
309 for (int y = 0; y < _height; ++y) {
310 std::vector<Color> row(_width, Color(0, 0, 0));
311 for (int x = 0; x < _width; ++x) {
312 const double checks = checksPerPixel[static_cast<std::size_t>(y) * static_cast<std::size_t>(_width) + static_cast<std::size_t>(x)];
313 double normalized = 0.0;
314 // Normalize across observed range. If all pixels have the same
315 // count (max==min) fall back to checks/maxChecks if possible,
316 // otherwise zero to show black.
317 if (maxChecks > minChecks + 1e-12) {
318 normalized = (checks - minChecks) / (maxChecks - minChecks);
319 } else if (maxChecks > 1e-12) {
320 normalized = checks / maxChecks;
321 } else {
322 normalized = 0.0;
323 }
324 row[x] = checksToHeatColor(normalized).clamp();
325 }
326
327 std::unique_lock<std::shared_mutex> lock(FrameBufferMutex());
328 framebuffer[y] = std::move(row);
329 }
330
331 return framebuffer;
332 }
