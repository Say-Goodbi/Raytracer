113 /**
114 * @brief Evaluate radiance along a ray using a two-branch shading model.
115 *
116 * The branch is chosen by the material's PDF for its sampled direction:
117 *
118 * **Specular / glass branch** (`pdf ≈ 0`, delta BSDFs):
119 * @code
120 * L = Le + evaluate(ωi, ωo) * π * castRay(ωi)
121 * @endcode
122 * The bounce origin is offset by 1e-4 along `ωi` to avoid self-intersection.
123 *
124 * **Diffuse branch** (`pdf > 0`):
125 * @code
126 * L = Le + evaluate(n, ωo, n) * π * Σ light->computeLight(hit)
127 * @endcode
128 * Incoming radiance comes from direct lighting only — no recursive bounce.
129 *
130 * Steps:
131 * 1. Return black if the bounce budget is exhausted.
132 * 2. Find the nearest primitive intersection; return black on miss.
133 * 3. Query the material for emitted radiance (`Le`) and a bounce direction.
134 * 4. Compute `pdf` for that direction.
135 * 5. If `pdf < 1e-6` (specular/glass): recurse along the bounce direction.
136 * 6. Otherwise: accumulate direct light from every scene light.
137 * 7. Combine emitted + shaded term and return.
138 *
139 * @param ray Incoming ray to evaluate.
140 * @param scene Scene containing all primitives and lights.
141 * @param depth Remaining bounce budget; returns black when ≤ 0.
142 * @return Radiance estimate along @p ray.
143 */
144 Color Camera::castRay(const Geometry::Ray &ray, Scene &scene, int depth) {
145 if (depth <= 0)
146 return Color(0, 0, 0);
147 std::optional<Geometry::HitRecord> closest = scene.hit(ray);
148
149 if (!closest)
150 return Color(0, 0, 0); // No intersection: return black (AKA: background, maybe change to a blue background ?)
151
152 IMaterial *mat = closest->material;
153 Color emitted = mat->emission();
154
155 Geometry::Vector3D viewDir = (ray.direction * -1).normalize();
156 Geometry::Vector3D bounceDir = mat->sample(closest->normal, viewDir);
157 float p = mat->pdf(closest->normal, viewDir, bounceDir);
158
159 if (p < 1e-6f) {
160 // Specular / glass: pure recursive bounce, no direct lighting
161 Geometry::Point3D bounceOrigin(
162 closest->point.x + bounceDir.x * 1e-4f,
163 closest->point.y + bounceDir.y * 1e-4f,
164 closest->point.z + bounceDir.z * 1e-4f);
165 Color tint = mat->evaluate(closest->normal, viewDir, bounceDir) * M_PI;
166 return emitted + tint * castRay(Geometry::Ray(bounceOrigin, bounceDir), scene, depth - 1);
167 }
168
169 Color directLight(0, 0, 0);
170 for (const auto& light : scene.getLights())
171 directLight = directLight + light->computeLight(*closest, scene.getPrimitives());
172
173 Color albedo = mat->evaluate(closest->normal, viewDir, closest->normal) * M_PI;
174 return emitted + albedo * directLight;
175 }

177 /**
178 * @brief Render the scene into a 2D framebuffer.
179 *
180 * Each pixel is super-sampled with a deterministic stratified grid of
181 * `_sqrtSamples × _sqrtSamples` sub-pixel rays. Sample positions are
182 * fixed at the **center of each sub-cell** — no jitter, no random offset —
183 * keeping the overall render fully deterministic.
184 *
185 * Per-pixel steps:
186 * 1. Iterate over the `_sqrtSamples × _sqrtSamples` sub-cell grid.
187 * 2. Compute the sub-cell center in normalized screen space (u, v).
188 * 3. Generate and path-trace a ray via `generateRay()` + `castRay()`.
189 * 4. Average all sub-pixel radiance values and clamp to [0, 1].
190 *
191 * Output layout: `frame[y][x]`, y = 0 at the top, x = 0 at the left.
192 *
193 * @param scene Scene to render.
194 * @return 2D grid of clamped colors: `frame[height][width]`.
195 */
196 std::vector<std::vector<Color>> Camera::render(Scene& scene, std::vector<std::vector<Color>>& framebuffer) {
197 scene.prepareAccelerationStructure(_useBVH);
198
199 {
200 std::unique_lock<std::shared_mutex> lock(FrameBufferMutex());
201 // Ensure framebuffer is the correct size before incremental updates start.
202 if ((int)framebuffer.size() != _height) framebuffer.assign(_height, std::vector<Color>(_width));
203 else {
204 for (auto &row : framebuffer) {
205 if ((int)row.size() != _width) row.assign(_width, Color(0,0,0));
206 }
207 }
208 }
209
210 for (int y = 0; y < _height; y++) {
211 std::vector<Color> row(_width, Color(0, 0, 0));
212 for (int x = 0; x < _width; x++) {
213 Color accumulated(0, 0, 0);
214
215 for (int sy = 0; sy < _sqrtSamples; sy++) {
216 for (int sx = 0; sx < _sqrtSamples; sx++) {
217 double u = (x + (sx + 0.5) / _sqrtSamples) / (_width - 1);
218 double v = (y + (sy + 0.5) / _sqrtSamples) / (_height - 1);
219 Geometry::Ray ray = generateRay(u, v);
220 accumulated = accumulated + castRay(ray, scene, _maxDepth);
221 }
222 }
223
224 int totalSamples = _sqrtSamples * _sqrtSamples;
225 row[x] = (accumulated * (1.0 / totalSamples)).clamp();
226 }
227
228 {
229 std::unique_lock<std::shared_mutex> lock(FrameBufferMutex());
230 framebuffer[y] = std::move(row);
231 }
232 }
233 return framebuffer;
234 }
