65 /**
66 * @brief Evaluate the rendering equation along a ray, deterministically.
67 *
68 * Implements the standard Monte Carlo path-tracing estimator:
69 * @code
70 * L(x, ωo) = Le(x, ωo) + ∫ f(x, ωi, ωo) Li(x, ωi) cosθi dωi
71 * @endcode
72 * but evaluates the integral at a **single deterministic direction** supplied
73 * by `IMaterial::sample()` instead of drawing a random sample. The
74 * estimator is then:
75 * @code
76 * L ≈ Le + f(ωi, ωo) * Li(ωi) * cosθi / pdf(ωi)
77 * @endcode
78 * This removes noise and makes rendering fully reproducible at the cost of
79 * variance reduction that random multi-sampling would otherwise provide.
80 *
81 * Steps:
82 * 1. Return black if the bounce budget is exhausted.
83 * 2. Find the nearest primitive intersection; return black on miss.
84 * 3. Query the material for its emitted radiance (`Le`).
85 * 4. Ask the material for a deterministic bounce direction (`ωi`).
86 * 5. If `pdf(ωi) ≈ 0`, return only the emitted term (degenerate surface).
87 * 6. Recurse along `ωi` (with a small normal-bias offset to avoid
88 * self-intersection) to get incoming radiance `Li`.
89 * 7. Combine: `Le + brdf * Li * cosθ / pdf`.
90 *
91 * @param ray Incoming ray to evaluate.
92 * @param scene Scene containing all primitives.
93 * @param depth Remaining bounce budget; returns black when ≤ 0.
94 * @return Radiance estimate along @p ray.
95 */
96 Color Camera::castRay(const Geometry::Ray &ray, Scene &scene, int depth) {
97 if (depth <= 0)
98 return Color(0, 0, 0);
99 std::optional<Geometry::HitRecord> closest;
100
101 for (const auto &primitive : scene.getPrimitives()) {
102 std::optional<Geometry::HitRecord> hit = primitive->hit(ray);
103 if (hit && (!closest || hit->rayDistance < closest->rayDistance))
104 closest = hit;
105 }
106 if (!closest)
107 return Color(0, 0, 0); // No intersection: return black (AKA: background, maybe change to a blue background ?)
108
109 IMaterial *mat = closest->material;
110 Color emitted = mat->emission();
111
112 Color directLight(0, 0, 0);
113 for (const auto& light : scene.getLights())
114 directLight = directLight + light->computeLight(*closest, scene.getPrimitives());
115
116 Geometry::Vector3D viewDir = (ray.direction * -1).normalize();
117
118 Geometry::Vector3D bounceDir = mat->sample(closest->normal, viewDir);
119 float p = mat->pdf(closest->normal, viewDir, bounceDir);
120
121 Color brdf = mat->evaluate(closest->normal, viewDir, bounceDir);
122
123 if (p < 1e-6f)
124 return emitted + brdf * directLight;
125
126 Geometry::Ray bounceRay(closest->point + closest->normal * 1e-4, bounceDir); // 1e-4 bias to avoid self-intersection
127 Color incoming = castRay(bounceRay, scene, depth - 1);
128 float cosTheta = std::max(0.0f, (float)closest->normal.dot(bounceDir));
129 return emitted + brdf * (directLight + incoming * (cosTheta / p));
130 }

132 /**
133 * @brief Render the scene into a 2D framebuffer.
134 *
135 * Each pixel is super-sampled with a deterministic stratified grid of
136 * `_sqrtSamples × _sqrtSamples` sub-pixel rays. Sample positions are
137 * fixed at the **center of each sub-cell** — no jitter, no random offset —
138 * keeping the overall render fully deterministic.
139 *
140 * Per-pixel steps:
141 * 1. Iterate over the `_sqrtSamples × _sqrtSamples` sub-cell grid.
142 * 2. Compute the sub-cell center in normalized screen space (u, v).
143 * 3. Generate and path-trace a ray via `generateRay()` + `castRay()`.
144 * 4. Average all sub-pixel radiance values and clamp to [0, 1].
145 *
146 * Output layout: `frame[y][x]`, y = 0 at the top, x = 0 at the left.
147 *
148 * @param scene Scene to render.
149 * @return 2D grid of clamped colors: `frame[height][width]`.
150 */
151 std::vector<std::vector<Color>> Camera::render(Scene& scene) {
152 std::vector<std::vector<Color>> framebuffer(_height, std::vector<Color>(_width));
153
154 for (int y = 0; y < _height; y++) {
155 for (int x = 0; x < _width; x++) {
156 Color accumulated(0, 0, 0);
157
158 for (int sy = 0; sy < _sqrtSamples; sy++) {
159 for (int sx = 0; sx < _sqrtSamples; sx++) {
160 double u = (x + (sx + 0.5) / _sqrtSamples) / (_width - 1);
161 double v = (y + (sy + 0.5) / _sqrtSamples) / (_height - 1);
162 Geometry::Ray ray = generateRay(u, v);
163 accumulated = accumulated + castRay(ray, scene, _maxDepth);
164 }
165 }
166
167 int totalSamples = _sqrtSamples * _sqrtSamples;
168 framebuffer[y][x] = (accumulated * (1.0 / totalSamples)).clamp();
169 }
170 }
171 return framebuffer;
172 }
