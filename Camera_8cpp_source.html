29 /**
30 * @brief Cast a ray into the scene and compute the resulting pixel color.
31 *
32 * Algorithm:
33 * 1. Test the ray against all primitives to find the closest intersection
34 * 2. If no intersection, return black (0, 0, 0)
35 * 3. If intersection found:
36 * a. Check if the hit point is in shadow by casting rays to all lights
37 * b. Compute color using the hit material's shader
38 * c. Darken by 90% if in shadow
39 * 4. Clamp color components to [0, 1] range
40 *
41 * @param ray The ray to cast into the scene
42 * @param scene Reference to the scene containing primitives and lights
43 * @return Resulting pixel color (R, G, B)
44 */
45 Color Camera::castRay(Geometry::Ray& ray, Scene& scene)
46 {
47 std::optional<Geometry::HitRecord> closest;
48 IPrimitive* hitPrimitive = nullptr;
49
50 // Find the closest intersection point along the ray
51 for (const auto &primitive : scene.getPrimitives()) {
52 std::optional<Geometry::HitRecord> hit = primitive->hit(ray);
53 if (hit && (!closest || hit->rayDistance < closest->rayDistance)) {
54 closest = hit;
55 hitPrimitive = primitive;
56 }
57 }
58
59 if (!closest)
60 return Color(0, 0, 0); // No intersection: return black
61
62 // Compute shadow and base material color
63 bool inShadow = computeShadow(closest.value(), scene);
64
65 Color finalColor = hitPrimitive->getMaterial()->computeColor(closest.value(), scene.getLights());
66
67 if (inShadow)
68 finalColor = finalColor * 0.1f; // Darken by 90% if in shadow
69
70 return finalColor.clamp();
71 }

73 /**
74 * @brief Determine if a hit point is in shadow.
75 *
76 * For each light in the scene:
77 * 1. Compute the direction from the hit point toward the light
78 * 2. Create a shadow ray from the hit point (with small epsilon offset to avoid self-intersection)
79 * 3. Check if any primitive blocks the ray before reaching the light
80 * 4. If any primitive blocks, the point is in shadow
81 *
82 * Shadow bias (0.001f): Small offset along surface normal to prevent self-shadowing artifacts
83 *
84 * @param hit The hit record at the intersection point
85 * @param scene Reference to the scene containing primitives and lights
86 * @return true if the hit point is in shadow from any light, false otherwise
87 */
88 bool Camera::computeShadow(Geometry::HitRecord &hit, Scene &scene)
89 {
90 for (const auto &light : scene.getLights()) {
91 Geometry::Vector3D toLight = light->getPosition() - hit.point;
92 float distanceToLight = toLight.length();
93 Geometry::Vector3D direction = toLight.normalize();
94
95 // Use small epsilon offset (shadow bias) to avoid self-intersection artifacts
96 Geometry::Point3D shadowOrigin = hit.point + direction * 0.001f;
97 Geometry::Ray shadowRay(shadowOrigin, direction);
98 for (const auto &primitive : scene.getPrimitives()) {
99 std::optional<Geometry::HitRecord> shadowHit = primitive->hit(shadowRay);
100 if (shadowHit && shadowHit->rayDistance < distanceToLight)
101 return true; // Primitive blocks light: in shadow
102 }
103 }
104 return false; // No primitives block light: not in shadow
105 }
