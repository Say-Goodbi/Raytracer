21 std::unique_ptr<Scene> SceneLoader::loadScene(void)
22 {
23 std::unique_ptr<Scene> scene = std::make_unique<Scene>();
24
25 if (!this->_parser._node)
26 throw RayTracer::ParsingException("Failed to parse scene configuration file.");
27 for (auto &[componentKey, creator] : _componentCreators) {
28 auto it = std::get<Object>(_parser._node->value).find(componentKey);
29 if (it == std::get<Object>(_parser._node->value).end())
30 throw RayTracer::ParsingException("Missing required component in scene configuration: " + componentKey);
31 creator(it->second, _pluginManager, scene);
32 }
33 return scene;
34 }
