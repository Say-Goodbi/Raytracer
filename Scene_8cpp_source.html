26 std::optional<Geometry::HitRecord> findClosestLinear(const Geometry::Ray &ray, const std::vector<std::shared_ptr<RayTracer::APrimitive>> &primitives)
27 {
28 std::optional<Geometry::HitRecord> closest;
29
30 for (const auto &primitive : primitives)
31 {
32 std::optional<Geometry::HitRecord> hit = primitive->hit(ray);
33 if (hit && (!closest || hit->rayDistance < closest->rayDistance))
34 closest = hit;
35 }

65 void Scene::prepareAccelerationStructure()
66 {
67 this->_bvh.reset();
68 this->_unboundedPrimitives.clear();
69
70 std::vector<std::shared_ptr<APrimitive>> boundedPrimitives;
71 boundedPrimitives.reserve(this->_primitives.size());
72
73 for (const auto &primitive : this->_primitives)
74 {
75 if (primitive->getBounds())
76 boundedPrimitives.push_back(primitive);
77 else
78 this->_unboundedPrimitives.push_back(primitive);
79 }
80
81 this->_bvh.emplace(boundedPrimitives);
82 }

84 std::optional<Geometry::HitRecord> Scene::hit(const Geometry::Ray &ray) const
85 {
86 std::optional<Geometry::HitRecord> closest;
87
88 if (this->_bvh)
89 closest = this->_bvh->hit(ray);
90
91 std::optional<Geometry::HitRecord> linearHit = findClosestLinear(ray, this->_unboundedPrimitives);
92 if (linearHit && (!closest || linearHit->rayDistance < closest->rayDistance))
93 closest = linearHit;
94
95 return closest;
96 }

98 void Scene::addPrimitives(RayTracer::NodePtr &primitives, PluginManager &pluginManager)
99 {
100 for (auto &[key, nodePtr] : std::get<Object>(primitives->value))
101 {
102 VectorObject &primitive = std::get<VectorObject>(nodePtr->value);

106

113 this->addPrimitive(prim);
