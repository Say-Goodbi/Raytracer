6 /**
7 * @brief Test ray–cylinder intersection using the quadratic form of the cylinder equation.
8 *
9 * Algorithm: Solve for intersection parameter t along the ray.
10 * Ray equation: P(t) = ray.origin + t * ray.direction
11 * Infinite cylinder equation: |P - _origin - ((P - _origin) · axis) * axis|² = _radius²
12 *
13 * Decompose the ray and delta into components perpendicular to the cylinder axis:
14 * aVec = ray.direction - (ray.direction · axis) * axis (perpendicular direction)
15 * bVec = delta - (delta · axis) * axis (perpendicular offset)
16 *
17 * This yields a quadratic equation in t:
18 * A*t² + B*t + C = 0
19 * where:
20 * A = aVec · aVec
21 * B = 2 * (aVec · bVec)
22 * C = bVec · bVec - radius²
23 *
24 * Steps:
25 * 1. Normalize the cylinder axis
26 * 2. Compute delta = ray.origin - _origin
27 * 3. Project ray direction and delta onto the axis to get perpendicular components
28 * 4. If A < 1e-10: ray is parallel to the cylinder axis (no lateral intersection)
29 * 5. Compute discriminant = B² - 4*A*C
30 * - If disc < 0: ray misses the cylinder entirely
31 * 6. Try nearest root t = (-B - sqrt(disc)) / (2*A); fall back to far root if t < 1e-4
32 * 7. Compute hit point and outward normal as the radial unit vector from the axis point
33 * 8. Return HitRecord with intersection point, normal, distance, and material
34 *
35 * @param ray The ray to test against the cylinder
36 * @return Optional HitRecord if intersection occurs with t >= 1e-4, empty otherwise
37 */
38 std::optional<Geometry::HitRecord> Cylinder::hit(const Geometry::Ray& ray) const
39 {
40 Geometry::Vector3D axis = _axis.normalize();
41 Geometry::Vector3D delta = ray.origin - _origin;
42

44 double dAxisDelta = axis.dot(delta);

63 }

65 double proj = axis.dot(hitPoint - _origin);
66 Geometry::Point3D axisPoint = _origin + axis * proj;
67 Geometry::Vector3D outwardNormal = (hitPoint - axisPoint) / static_cast<double>(_radius);
68 return Geometry::HitRecord(hitPoint, outwardNormal, t, this->getMaterial());
