20 * @brief Render the scene and write the result to a PPM file.
21 *
22 * Output format: PPM (Portable Pixmap) P3 (ASCII) format
23 * PPM structure:
24 * - Header: "P3\n"
25 * - Dimensions: "width height\n"
26 * - Max color value: "255\n"
27 * - Pixel data: R G B values (0-255) per line, one pixel per line
28 *
29 * Algorithm:
30 * 1. Call renderer->render(scene) to get the framebuffer
31 * 2. Extract framebuffer dimensions (height × width)
32 * 3. Open output file for writing
33 * 4. Write PPM header with dimensions
34 * 5. Iterate through each pixel [y][x] and write RGB values (converted to 0-255 range)
35 * 6. Close the file
36 *
37 * @param scene The scene to render and write
38 */
39 void SceneWriter::execute(Scene& scene)
40 {
41 std::vector<std::vector<RayTracer::Color>> frameBuffer = _renderer->render(scene);
42 if (frameBuffer.empty())
43 return;
44
45 int height = static_cast<int>(frameBuffer.size());
46 int width = static_cast<int>(frameBuffer[0].size());
47
48 std::ofstream ofs(_outputFile, std::ios::out);
49 if (!ofs)
50 return;
51
52 // Write PPM header
53 ofs << "P3\n" << width << " " << height << "\n255\n";
54
55 // Write pixel data: iterate y (row) then x (column)
56 for (int y = 0; y < height; ++y) {
57 for (int x = 0; x < width; ++x) {
58 const Color &c = frameBuffer[y][x];
59 // Convert color from [0,1] to [0,255] range
60 ofs << static_cast<int>(c.r * 255) << " "
61 << static_cast<int>(c.g * 255) << " "
62 << static_cast<int>(c.b * 255) << "\n";
63 }
64 }
