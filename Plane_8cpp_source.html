5 /**
6 * @brief Test ray–plane intersection using the algebraic plane equation.
7 *
8 * Algorithm: Solve for intersection parameter t along the ray.
9 * Ray equation: P(t) = ray.origin + t * ray.direction
10 * Plane equation: (P - _point) · _normal = 0
11 *
12 * Substituting the ray into the plane equation:
13 * t = ((_point - ray.origin) · _normal) / (ray.direction · _normal)
14 *
15 * Steps:
16 * 1. Compute denominator = ray.direction · _normal
17 * - If |denominator| < 1e-6: ray is parallel to plane (no intersection)
18 * 2. Compute numerator = (_point - ray.origin) · _normal
19 * 3. Solve t = numerator / denominator
20 * 4. If t > 1e-4: intersection is in front of ray origin (epsilon avoids self-intersection)
21 * 5. Flip normal toward the incoming ray (for correct back-face shading)
22 * 6. Return HitRecord with intersection point, normal, distance, and material
23 *
24 * @param ray The ray to test against the plane
25 * @return Optional HitRecord if intersection occurs with t >= 0, empty otherwise
26 */
27 std::optional<Geometry::HitRecord> Plane::hit(const Geometry::Ray& ray) const
28 {

30 if (std::abs(denom) > 1e-6) {

33 Geometry::Vector3D outwardNormal = denom < 0 ? _normal : _normal * -1;
