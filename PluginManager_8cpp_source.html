25void RayTracer::PluginManager::loadPlugins()
26{
27 _plugins.clear();
28 struct dirent* entry = nullptr;
29 DIR* dir = opendir(_pluginDirectory.c_str());
30 if (!dir)
31 throw RayTracer::Exception("Failed to open plugin directory: " + _pluginDirectory);
32 while ((entry = readdir(dir)) != nullptr) {
33 std::string filename(entry->d_name);
34 if (filename.size() > 3 && filename.substr(filename.size() - 3) == ".so") {
35 std::string path = _pluginDirectory + "/" + filename;
36 loadPlugin(path);
37 }
38 }
39 if (_plugins.find("CorePlugin") == _plugins.end())
40 std::cerr << "Warning: CorePlugin not found. The application may not function correctly without it." << std::endl;
41 closedir(dir);
42}

44bool RayTracer::PluginManager::loadPlugin(const std::string& path)
45{
46 void* handle = dlopen(path.c_str(), RTLD_LAZY);
47 if (!handle) {
48 std::cerr << "Failed to load plugin: " << path << " - " << dlerror() << std::endl;
49 return false;
50 }
51 using GetNameFn = std::string (*)();
52 using InitializersMap = std::map<std::string, std::function<RayTracer::Component(RayTracer::NodePtr)>>;
53 using GetInitializersFn = InitializersMap (*)();
54
55 GetNameFn getName = reinterpret_cast<GetNameFn>(dlsym(handle, "getName"));
56 GetInitializersFn getInitializers = reinterpret_cast<GetInitializersFn>(dlsym(handle, "getInitializers"));
57
58 if (getName == nullptr || getInitializers == nullptr) {
59 std::cerr << "Failed to find plugin functions in: " << path << " - " << dlerror() << std::endl;
60 dlclose(handle);
61 return false;
62 }
63 std::string pluginName = getName();
64 if (_plugins.find(pluginName) != _plugins.end()) {
65 std::cerr << "Plugin with name " << pluginName << " already loaded. Skipping: " << path << std::endl;
66 dlclose(handle);
67 return false;
68 }
69 InitializersMap initializers = getInitializers();
70 overrideInitializers(initializers);
71 _plugins[pluginName] = initializers;
72 _handles.push_back(handle);
73 return true;
74}

91{
92 for (const auto& pluginPair : _plugins) {
